bootstrap_ci_sequential <- function(data, n_bootstrap = 1000, confidence = 0.95) {
  boot_means <- numeric(n_bootstrap)
  
  for (i in 1:n_bootstrap) {
    boot_sample <- sample(data, size = length(data), replace = TRUE)
    boot_means[i] <- mean(boot_sample)
  }
  
  alpha <- 1 -confidence
  
  ci <- quantile(
    boot_means,
    probs = c(alpha / 2, 1 - alpha / 2)
  )
  names(ci) <- c("lower", "upper")
  return(ci)
}

library(parallel)
bootstrap_ci_parallel <- function(data, n_bootstrap = 1000, confidence = 0.95) {
  boot_means <- mclapply(
    1:n_bootstrap,
    function(i) {
      boot_sample <- sample(data, size = length(data), replace = TRUE)
      mean(boot_sample)
    },
    mc.cores = safe_cores
  )
  boot_means <- unlist(boot_means)
  alpha <- 1 - confidence
  ci <- quantile(
    boot_means,
    probs = c(alpha / 2, 1 - alpha/2)
  )
  
  names(ci) <- c("lower", "upper")
  return(ci)
}
